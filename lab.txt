# FORM1.CS

```csharp
using System;
using System.Data;
using System.Data.SqlClient;
using System.IO;
using System.Windows.Forms;
using iTextSharp.text;
using iTextSharp.text.pdf;

namespace LabSistemi
{
    public partial class Form1 : Form
    {
        SqlConnection baglanti =
        new SqlConnection(@"Server=.\SQLEXPRESS;
        Database=LabSistemi;
        Trusted_Connection=True;");

        public Form1()
        {
            InitializeComponent();

            cmbOnay.Items.Add("Onaylandı");
            cmbOnay.Items.Add("Bekliyor");
            cmbOnay.Items.Add("Reddedildi");
        }

        void Listele()
        {
            SqlDataAdapter da =
            new SqlDataAdapter("SELECT * FROM Hastalar", baglanti);

            DataTable dt = new DataTable();
            da.Fill(dt);

            dataGridView1.DataSource = dt;
        }

        private void btnKaydet_Click(object sender, EventArgs e)
        {
            try
            {
                baglanti.Open();

                SqlCommand komut = new SqlCommand(
                "INSERT INTO Hastalar " +
                "(AdSoyad,TC,Tahlil,Sonuc,DoktorOnay,Tarih) " +
                "VALUES (@a,@b,@c,@d,@e,@f)", baglanti);

                komut.Parameters.AddWithValue("@a", txtAdSoyad.Text);
                komut.Parameters.AddWithValue("@b", txtTC.Text);
                komut.Parameters.AddWithValue("@c", txtTahlil.Text);
                komut.Parameters.AddWithValue("@d", txtSonuc.Text);
                komut.Parameters.AddWithValue("@e", cmbOnay.Text);
                komut.Parameters.AddWithValue("@f", DateTime.Now);

                komut.ExecuteNonQuery();

                MessageBox.Show("Kayıt Eklendi");

                baglanti.Close();

                Listele();
            }
            catch (Exception ex)
            {
                MessageBox.Show(ex.Message);
            }
        }

        private void btnListele_Click(object sender, EventArgs e)
        {
            Listele();
        }

        private void btnAra_Click(object sender, EventArgs e)
        {
            SqlDataAdapter da = new SqlDataAdapter(
            "SELECT * FROM Hastalar WHERE AdSoyad LIKE '%"
            + txtAra.Text + "%'", baglanti);

            DataTable dt = new DataTable();
            da.Fill(dt);

            dataGridView1.DataSource = dt;
        }

        private void btnSil_Click(object sender, EventArgs e)
        {
            try
            {
                baglanti.Open();

                SqlCommand komut =
                new SqlCommand("DELETE FROM Hastalar WHERE Id=@id",
                baglanti);

                komut.Parameters.AddWithValue("@id",
                dataGridView1.CurrentRow.Cells[0].Value);

                komut.ExecuteNonQuery();

                baglanti.Close();

                MessageBox.Show("Kayıt Silindi");

                Listele();
            }
            catch (Exception ex)
            {
                MessageBox.Show(ex.Message);
            }
        }

        private void btnPDF_Click(object sender, EventArgs e)
        {
            SaveFileDialog sfd = new SaveFileDialog();

            sfd.Filter = "PDF Dosyası|*.pdf";

            if (sfd.ShowDialog() == DialogResult.OK)
            {
                Document doc = new Document();

                PdfWriter.GetInstance(doc,
                new FileStream(sfd.FileName, FileMode.Create));

                doc.Open();

                doc.Add(new Paragraph("Laboratuvar Sonuç Raporu"));
                doc.Add(new Paragraph(" "));

                doc.Add(new Paragraph("Hasta : "
                + txtAdSoyad.Text));

                doc.Add(new Paragraph("TC : "
                + txtTC.Text));

                doc.Add(new Paragraph("Tahlil : "
                + txtTahlil.Text));

                doc.Add(new Paragraph("Sonuç : "
                + txtSonuc.Text));

                doc.Add(new Paragraph("Doktor Onay : "
                + cmbOnay.Text));

                doc.Close();

                MessageBox.Show("PDF Oluşturuldu");
            }
        }

        private void çıkışToolStripMenuItem_Click(object sender, EventArgs e)
        {
            Application.Exit();
        }

        private void yardımToolStripMenuItem_Click(object sender, EventArgs e)
        {
            MessageBox.Show(
            "Bu program laboratuvar sonuçlarını kayıt etmek için yapılmıştır.");
        }

        private void hakkındaToolStripMenuItem_Click(object sender, EventArgs e)
        {
            MessageBox.Show(
            "Laboratuvar Sonuç Sistemi\nVersiyon 1.0\n2026");
        }
    }
}
```
